183  void decision_tree<T, U>::add_sequence(std::shared_ptr<decision_value<T, U> > dec_val, bool reverse_order)
184  {
185  std::shared_ptr<T> first_item;

200  root_->add_sequence(dec_val, 0, reverse_order);

202  else
