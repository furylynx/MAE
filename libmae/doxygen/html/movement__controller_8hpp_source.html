307  {

313  if (imd_)
314  {
315  std::shared_ptr<U> sequence = imd_->detect_movement(timestamp, framerate_, skeleton, body_parts_);
316  current_sequence_ = sequence;
317  notify_sequence_listeners(timestamp, sequence);

319  if (isr_ != nullptr)
320  {
321  current_recognition_ = isr_->recognize_sequence(framerate_, sequence, body_parts_);
322

324  {
325  notify_recognition_listeners(timestamp, current_recognition_);
326  }
327  }
328  else
329  {
330  throw std::invalid_argument("No sequence recognizer defined in the movement controller.");
331  }
332  }

335  throw std::invalid_argument("No movement detector defined in the movement controller.");
