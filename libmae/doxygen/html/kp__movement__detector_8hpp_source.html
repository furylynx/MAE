186  template<typename T, typename U>
187  std::shared_ptr<U> kp_movement_detector<T, U>::detect_movement(long timestamp, double framerate, std::shared_ptr<T> skeleton,
188  std::vector<bone> body_parts)
189  {
190  if (debug_)
191  {
192  std::cout << "kp_movement_detector: detect movement" << std::endl;
193  }
194 
195  std::shared_ptr<U> sequence;
196 
197  if (ipd_)
198  {
199  std::shared_ptr<general_pose> pose = ipd_->pose(skeleton, body_parts, previous_pose_);
200 
201  //update previous pose
202  previous_pose_ = pose;
203 
204  //notify pose listeners
205  notify_listeners(timestamp, pose);
206 
207  if (ikpd_)
208  {
209  std::shared_ptr<general_enriched_pose> enriched_pose = ikpd_->estimate_frame(pose, poses_, body_parts);
210 
211  poses_.push_front(enriched_pose);
212  if (poses_.size() > pose_buffer_size_)
213  {
214  poses_.pop_back();
215  }
216 
217  if (isg_)
218  {
219  sequence = isg_->generate_sequence(framerate, poses_, body_parts);
220  }
221  }
222  }
223 
224  return sequence;
225  }
