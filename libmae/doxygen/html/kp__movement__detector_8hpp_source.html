195  template<typename T, typename U>
196  std::shared_ptr<U> kp_movement_detector<T, U>::detect_movement(long timestamp, double framerate,
197  std::shared_ptr<T> skeleton, std::vector<bone> body_parts)
198  {
199  if (debug_)
200  {
201  std::cout << "kp_movement_detector: detect movement" << std::endl;
202  }
203 
204  std::shared_ptr<U> sequence;
205 
206  if (ipd_)
207  {
208  std::shared_ptr<general_pose> pose = ipd_->pose(framerate, skeleton, body_parts, previous_pose_);
209 
210  //update previous pose
211  previous_pose_ = pose;
212 
213  //notify pose listeners
214  notify_listeners(timestamp, pose);
215 
216  if (ikpd_)
217  {
218  std::shared_ptr<general_enriched_pose> enriched_pose = ikpd_->estimate_frame(framerate, pose, poses_, body_parts);
219 
220  poses_.push_front(enriched_pose);
221  if (poses_.size() > pose_buffer_size_)
222  {
223  poses_.pop_back();
224  }
225 
226  if (isg_)
227  {
228  sequence = isg_->generate_sequence(framerate, poses_, body_parts);
229  }
230  }
231  }
232 
233  return sequence;
234  }
