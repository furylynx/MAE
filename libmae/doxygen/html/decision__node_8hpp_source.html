225  template<typename T, typename U>
226  void decision_node<T, U>::add_sequence(std::shared_ptr<decision_value<T, U> > decision_value, int step,
227  bool reverse_order)
228  {
229  std::vector<std::shared_ptr<T> > sequence = decision_value->get_sequence();
230 
231  if (step == sequence.size() - 1)
232  {
233  //last element in sequence
234  //we can safely add the sequence, since the check for same movements is done in the parent node
235  values_.push_back(decision_value);
236  }
237  else
238  {
239  //next index for normal or reverse order
240  unsigned int current_index = step;
241  unsigned int next_index = step + 1;
242  if (reverse_order)
243  {
244  current_index = sequence.size() - 1 - step;
245  next_index = sequence.size() - 1 - (step + 1);
246  }
247 
248  std::vector<std::shared_ptr<decision_node<T, U> > > matching_children = get_matching_children(sequence.at(next_index),
249  sequence.at(current_index), true);
250 
251  if (matching_children.size() > 0)
252  {
253  //use the first possible match to insert
254  matching_children.at(0)->add_sequence(decision_value, step + 1, reverse_order);
255  }
256  else
257  {
258  //no child had a match, therefore a new child has to be added
259  children_.push_back(
260  std::shared_ptr<decision_node<T, U> >(
261  new decision_node(decision_maker_, sequence.at(next_index))));
262  children_.back()->add_sequence(decision_value, step + 1, reverse_order);
263  }
264  }

351  template<typename T, typename U>
352  std::vector<std::shared_ptr<decision_value<T, U> > > decision_node<T, U>::find_submatches(
353  std::vector<std::shared_ptr<T> > whole_sequence, int step, int end_pos, bool reverse_order)
354  {
355  if (end_pos < 0)
356  {
357  //set end_pos if not specified
358  end_pos = whole_sequence.size() - 1;
359  }
360 
361  //all treasures that are given by submatches for the whole sequence
362  std::vector<std::shared_ptr<decision_value<T, U> > > result;
363 
364  unsigned int current_index = step;
365  unsigned int next_index = step + 1;
366 
367  if (reverse_order)
368  {
369  //change indices for reverse order
370  current_index = whole_sequence.size() - 1 - step;
371  next_index = whole_sequence.size() - 1 - (step + 1);
372  }
373 
374  if (!is_leaf() && step != end_pos)
375  {
376  //recursively search all matching children for sequences
377  std::vector<std::shared_ptr<decision_node<T, U> > > matching_children = get_matching_children(
378  whole_sequence.at(next_index), whole_sequence.at(current_index), false);
379 
380  for (unsigned int i= 0; i < matching_children.size(); i++)
381  {
382  //recurse; step is just incremented (it is not equal to the sequence element's index - this is next_index)
383  std::vector<std::shared_ptr<decision_value<T, U> > > child_result = matching_children.at(i)->find_submatches(whole_sequence, step + 1, end_pos, reverse_order);
384  result.insert(result.end(), child_result.begin(), child_result.end());
385  }
386  }
387 
388  //just insert all values for this node (since node had a match)
389  for (unsigned int i = 0; i < values_.size(); i++)
390  {
391  result.push_back(values_.at(i));
392  }
393 
394  return result;
395  }

398  std::vector<std::shared_ptr<decision_value<T, U> > > decision_node<T, U>::find_matches(
399  std::vector<std::shared_ptr<T> > sequence, int step, int end_pos)
400  {
401  if (end_pos < 0)
402  {
403  end_pos = sequence.size() - 1;
404  }
405 
406  if (step == sequence.size() - 1)
407  {
408  return values_;
409  }
410  else
411  {
412  //all treasures that are given by matches for the sequence
413  std::vector<std::shared_ptr<decision_value<T, U> > > result;
414 
415  if (!is_leaf())
416  {
417  if (step != end_pos)
418  {
419  //recursively search all matching children for sequences - since insertion matching is equal to the exact matching
420  std::vector<std::shared_ptr<decision_node<T, U> > > matching_children = get_matching_children(
421  sequence.at(step + 1), sequence.at(step), true);
422 
423  for (unsigned int i= 0; i < matching_children.size(); i++)
424  {
425  std::vector<std::shared_ptr<decision_value<T, U> > > child_result = matching_children.at(i)->find_matches(sequence, step + 1, end_pos);
426  result.insert(result.end(), child_result.begin(), child_result.end());
427  }
